## example
```javascript
const hydrus = require("hydrus.js")
var client = new hydrus({
  key: 'eb79e9e2a40dddd159299408db072ed6e6573d488b50db499f70a211ba1fa1da',
});
client.api_version((response) => {
  console.log(response); // returns { version: 9 }
});
```
